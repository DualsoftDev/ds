```
 [sys]나의_취미  = {
    동전넣기 > 위치확인 > 버튼;

      동전넣기 = {뽑기.동전인식 ~ _ }
      버튼     = {장치.버튼누름 ~ _ }

    (X축_덜감확인) > @pushs (좌);
    (X축_더감확인) > @pushs (우);
    (Y축_덜감확인) > @pushs (상);
    (Y축_더감확인) > @pushs (하);
    (고양이.야옹) > Loㄱ;

      좌 = {뽑기.X- ~ _ };
      우 = {뽑기.X+ ~ _ };
      상 = {뽑기.Y+ ~ _ };
      하 = {뽑기.Y- ~ _ };

  }

  [sys]고양이  = {

    (움직임감지) > 야옹;
    야옹 = {성대울림 ~ 소리야옹};

  }

  
  [sys]뽑기  = {
    X- <||> X+;
    Y- <||> Y+;
    Z- <||> Z+;
    풀기 <||> 잡기;
    잡기버튼 > 인형뽑기처리 <| _runr;
    인형뽑기처리 = {풀기, Z+ > 잡기 > Z- > X-, Y-, 풀기 > 잡기};

    X- = {장치.출력X- ~ 장치.센서X-};
    X+ = {장치.출력X+ ~ 장치.센서X+};
    
    Y- = {장치.출력Y- ~ 장치.센서Y-};
    Y+ = {장치.출력Y+ ~ 장치.센서Y+};
    
    Z- = {장치.출력Z- ~ 장치.센서Z-};
    Z+ = {장치.출력Z+ ~ 장치.센서Z+};
    
    잡기 = {장치.출력잡기 ~ @1sec};
    풀기 = {장치.출력풀기 ~ 장치.풀기센서};

    버튼 = {장치.결정 ~ _};

  }

  [sys]장치  = { 출력X-;출력X+;출력Y-;출력Y+;출력Z-;출력Z+;
                 센서-; 센서+; 센서Y-;센서Y+;센서Z-;센서Z+;
                 출력잡기;출력풀기;풀기센서;결정;
  }
```
