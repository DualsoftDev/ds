```
 [Sys]나의_취미  = {
    동전넣기 > @s(30) > 뽑기.잡기버튼;
    버튼누름 > 뽑기.잡기버튼;

    (X축_덜감확인) > @pushs (뽑기.X+);
    (X축_더감확인) > @pushs (뽑기.X-);
    (Y축_덜감확인) > @pushs (뽑기.Y+);
    (Y축_더감확인) > @pushs (뽑기.Y-);
    (고양이.야옹) > Loㄱ;

  }

  [Sys]고양이  = {

    (움직임감지) > 야옹;
    야옹 = {성대울림 ~ 소리야옹};

  }

  
  [Sys]뽑기  = {
    X- <||> X+;
    Y- <||> Y+;
    Z- <||> Z+;
    풀기 <||> 잡기;
    잡기버튼 > 인형뽑기처리 <| _rr;
    인형뽑기처리 = {풀기, Z+ > 잡기 > Z- > X-, Y-, 풀기 > 잡기};

    X- = {부품.출력X- ~ 부품.센서X-};
    X+ = {부품.출력X+ ~ 부품.센서X+};
    
    Y- = {부품.출력Y- ~ 부품.센서Y-};
    Y+ = {부품.출력Y+ ~ 부품.센서Y+};
    
    Z- = {부품.출력Z- ~ 부품.센서Z-};
    Z+ = {부품.출력Z+ ~ 부품.센서Z+};
    
    잡기 = {부품.출력잡기 ~ @1sec};
    풀기 = {부품.출력풀기 ~ 부품.풀기센서};

    버튼 = {부품.결정 ~ _};

  }

  [Sys]PLC  = { 출력X-;출력X+;출력Y-;출력Y+;출력Z-;출력Z+;
                 센서-; 센서+; 센서Y-;센서Y+;센서Z-;센서Z+;
                 출력잡기;출력풀기;풀기센서;결정;
  }
```
